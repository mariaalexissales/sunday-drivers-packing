module Packing
{
    imports
    {
        Base,
        SoulForge
    }

    item 5pkCritMulti
    {
        Weight          = 0.05
        Type            = Normal,
        DisplayName     = 5 x Augment Critical Multiplier,
        Icon            = BusTicket,
        WorldStaticModel= PaperNapkins_Ground,
        DisplayCategory = SoulForge
    }

    item 10pkCritMulti
    {
        Weight          = 0.1
        Type            = Normal,
        DisplayName     = 10 x Augment Critical Multiplier,
        Icon            = BusTicket,
        WorldStaticModel= PaperNapkins_Ground,
        DisplayCategory = SoulForge
    }

    recipe Pack 5
    {
        destroy CritMultiTicket=5,
    
        Result  :5pkCritMulti,
        OnTest  :Recipe.OnTest.IsFavorite,
        Time    :5.0,
        Category:Storage
    }

    recipe Unpack 5
    {
        destroy 5pkCritMulti,

        Result  :CritMultiTicket=5,
        Time    :5.0,
        Category:Storage
    }

    recipe Pack 10
    {
        destroy CritMultiTicket=10,

        Result  :10pkCritMulti,
        OnTest  :Recipe.OnTest.IsFavorite,
        Time    :10.0,
        Category:Storage
    }

    recipe Unpack 10
    {
        destroy 10pkCritMulti,

        Result  :CritMultiTicket=10,
        Time    :10.0,
        Category:Storage
    }

    recipe Put Together
    {
        5pkCritMulti=2,

        Result  :10pkCritMulti=1,
        Time    :5.0
    }

    recipe Split
    {
        10pkCritMulti,

        Result  :5pkCritMulti=2,
        Time    :5.0
    }
}
