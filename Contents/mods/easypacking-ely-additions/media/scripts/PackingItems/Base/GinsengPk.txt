module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkGinseng
	{
		DisplayName		= 10 x Ginseng,
		Weight			= 0.5,
		Type			= Normal,
		Icon			= PkGinsengSmallBox,
		WorldStaticModel= SmallGinseng
	}
	
	item 40pkGinseng
	{
		DisplayName		= 40 x Ginseng,
		Weight			= 2.8,
		Type			= Normal,
		Icon			= PkGinsengMediumBox,
		WorldStaticModel= MediumGinseng
	}
	
	recipe Pack 10
	{
		destroy Ginseng=10,
		
		Result	:10pkGinseng,
		OnCreate:Recipe.OnCreate.SaveFood,
		OnTest	:Recipe.OnTest.IsFavorite,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkGinseng,
		
		Result	:Ginseng=10,
		OnCreate:Recipe.OnCreate.LoadFood,
		RemoveResultItem:true,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Pack 40
	{
		destroy Ginseng=40,
		
		Result	:40pkGinseng,
		OnCreate:Recipe.OnCreate.SaveFood,
		OnTest	:Recipe.OnTest.IsFavorite,
		Time	:22.5,
		Category:Storage,
	}
	
	recipe Unpack 40
	{
		destroy 40pkGinseng,
		
		Result	:Ginseng=40,
		OnCreate:Recipe.OnCreate.LoadFood,
		RemoveResultItem:true,
		Time	:22.5,
		Category:Storage,
	}

	recipe Put Together
	{
		10pkGinseng=4,
		
		Result	:40pkGinseng=1,
		Time	:7.5,
	}

	recipe Split in Four
	{
		40pkGinseng=1,
		
		Result	:10pkGinseng=4,
		Time	:7.5,
	}
}
