module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkNettles
	{
		DisplayName		= 10 x Nettles,
		Weight			= 0.5,
		Type			= Normal,
		Icon			= Nettle,
		WorldStaticModel= Nettles_Ground
	}
	
	item 40pkNettles
	{
		DisplayName		= 40 x Nettles,
		Weight			= 2.8,
		Type			= Normal,
		Icon			= Nettle,
		WorldStaticModel= Nettles_Ground
	}
	
	recipe Pack 10
	{
		destroy Nettles=10,
		
		Result	:10pkNettles,
		OnCreate:Recipe.OnCreate.SaveFood,
		OnTest	:Recipe.OnTest.IsHerb,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkNettles,
		
		Result	:Nettles=10,
		OnCreate:Recipe.OnCreate.LoadFood,
		RemoveResultItem:true,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Pack 40
	{
		destroy Nettles=40,
		
		Result	:40pkNettles,
		OnCreate:Recipe.OnCreate.SaveFood,
		OnTest	:Recipe.OnTest.IsHerb,
		Time	:22.5,
		Category:Storage,
	}
	
	recipe Unpack 40
	{
		destroy 40pkNettles,
		
		Result	:Nettles=40,
		OnCreate:Recipe.OnCreate.LoadFood,
		RemoveResultItem:true,
		Time	:22.5,
		Category:Storage,
	}

	recipe Put Together
	{
		10pkNettles=4,
		
		Result	:40pkNettles=1,
		Time	:7.5,
	}

	recipe Split in Four
	{
		40pkNettles=1,
		
		Result	:10pkNettles=4,
		Time	:7.5,
	}
}
