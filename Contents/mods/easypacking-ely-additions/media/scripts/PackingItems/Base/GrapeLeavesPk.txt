module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkGrapeLeaves
	{
		DisplayName		= 10 x Grape Leaves,
		Weight			= 0.5,
		Type			= Normal,
		Icon			= GrapeLeaves,
		WorldStaticModel= GrapeLeaves
	}
	
	item 50pkGrapeLeaves
	{
		DisplayName		= 50 x Grape Leaves,
		Weight			= 2.8,
		Type			= Normal,
		Icon			= GrapeLeaves,
		WorldStaticModel= GrapeLeaves
	}
	
	recipe Pack 10
	{
		destroy GrapeLeaves=10,
		
		Result	:10pkGrapeLeaves,
		OnCreate:Recipe.OnCreate.SaveFood,
		OnTest	:Recipe.OnTest.IsFavorite,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkGrapeLeaves,
		
		Result	:GrapeLeaves=10,
		OnCreate:Recipe.OnCreate.LoadFood,
		RemoveResultItem:true,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Pack 50
	{
		destroy GrapeLeaves=50,
		
		Result	:50pkGrapeLeaves,
		OnCreate:Recipe.OnCreate.SaveFood,
		OnTest	:Recipe.OnTest.IsFavorite,
		Time	:22.5,
		Category:Storage,
	}
	
	recipe Unpack 50
	{
		destroy 50pkGrapeLeaves,
		
		Result	:GrapeLeaves=50,
		OnCreate:Recipe.OnCreate.LoadFood,
		RemoveResultItem:true,
		Time	:22.5,
		Category:Storage,
	}

	recipe Put Together
	{
		10pkGrapeLeaves=5,
		
		Result	:50pkGrapeLeaves=1,
		Time	:7.5,
	}

	recipe Split in Five
	{
		50pkGrapeLeaves=1,

	}
