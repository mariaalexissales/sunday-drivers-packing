module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkleadpipe
	{
		Weight			= 7.5,
		Type			= Normal,
		DisplayName		= 10 x Lead Pipes,
		DisplayCategory = Ammo,
		Icon			= PkLeadPipeMediumBox,
		WorldStaticModel= MediumLeadPipe
	}
	
	item 50pkleadpipe
	{
		Weight			= 32.5,
		Type			= Normal,
		DisplayName		= 50 x Lead Pipes,
		DisplayCategory = Ammo,
		Icon			= PkLeadPipeBigBox,
		WorldStaticModel= BigLeadPipe
	}
	
	recipe Pack 10
	{
		destroy LeadPipe=10,
		
		Result:10pkleadpipe,
		OnTest:Recipe.OnTest.IsFavorite,
		Time:55.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkleadpipe,
		
		Result:LeadPipe=10,
		Time:55.0,
		Category:Storage,
	}
	

	recipe Put Together
	{
		10pkleadpipe=5,
		
		Result:50pkleadpipe=1,
		Time:37.5,
	}

	recipe Split in Five
	{
		50pkleadpipe=1,
		
		Result:10pkleadpipe=5,
		Time:37.5,
	}
}
