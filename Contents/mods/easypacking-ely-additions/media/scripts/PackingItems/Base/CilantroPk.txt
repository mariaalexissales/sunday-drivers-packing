module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkCilantro
	{
		DisplayName		= 10 x Cilantro,
		Weight			= 0.5,
		Type			= Normal,
		Icon			= HerbCilantro,
		WorldStaticModel= HerbCilantro_Ground
	}
	
	item 50pkCilantro
	{
		DisplayName		= 50 x Cilantro,
		Weight			= 2.8,
		Type			= Normal,
		Icon			= HerbCilantro,
		WorldStaticModel= HerbCilantro_Ground
	}
	
	recipe Pack 10
	{
		destroy Cilantro=10,
		
		Result	:10pkCilantro,
		OnCreate:Recipe.OnCreate.SaveFood,
		OnTest	:Recipe.OnTest.IsFavorite,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkCilantro,
		
		Result	:Cilantro=10,
		OnCreate:Recipe.OnCreate.LoadFood,
		RemoveResultItem:true,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Pack 50
	{
		destroy Cilantro=50,
		
		Result	:50pkCilantro,
		OnCreate:Recipe.OnCreate.SaveFood,
		OnTest	:Recipe.OnTest.IsFavorite,
		Time	:22.5,
		Category:Storage,
	}
	
	recipe Unpack 50
	{
		destroy 50pkCilantro,
		
		Result	:Cilantro=50,
		OnCreate:Recipe.OnCreate.LoadFood,
		RemoveResultItem:true,
		Time	:22.5,
		Category:Storage,
	}

	recipe Put Together
	{
		10pkCilantro=5,
		
		Result	:50pkCilantro=1,
		Time	:7.5,
	}

	recipe Split in Five
	{
		50pkCilantro=1,
		
		Result	:10pkCilantro=5,
		Time	:7.5,
	}
}
