module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkComfrey
	{
		Weight			= 0.5,
		Type			= Normal,
		DisplayName		= 10 x Comfrey,
		Icon			= PkComfreySmallBox,
		WorldStaticModel= SmallComfrey
	}
	
	item 40pkComfrey
	{
		Weight			= 2.8,
		Type			= Normal,
		DisplayName		= 40 x Comfrey,
		Icon			= PkComfreyMediumBox,
		WorldStaticModel= MediumComfrey
	}
	
	recipe Pack 10
	{
		destroy Comfrey=10,
		
		Result	:10pkComfrey,
		OnTest	:Recipe.OnTest.IsFavorite,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkComfrey,
		
		Result	:Comfrey=10,
		Time	:45.0,
		Category:Storage,
	}
	
	recipe Pack 40
	{
		destroy Comfrey=40,
		
		Result	:40pkComfrey,
		OnTest	:Recipe.OnTest.IsFavorite,
		Time	:22.5,
		Category:Storage,
	}
	
	recipe Unpack 40
	{
		destroy 40pkComfrey,
		
		Result	:Comfrey=40,
		Time	:22.5,
		Category:Storage,
	}

	recipe Put Together
	{
		10pkComfrey=4,
		
		Result	:10pkComfrey=1,
		Time	:7.5,
	}

	recipe Split in Four
	{
		40pkComfrey=1,
		
		Result	:10pkComfrey=4,
		Time	:7.5,
	}
}
